public TestClassDSL<T, C> {

	public String[] names;
	private final T password;
	protected C color;

	public TestClassDSL(String[] names, T password, C color) {
		this.names = names;
		this.password = password;
		this.color = color;
	}

	public String[] getNames() {
		return names;
	}

	public T getPassword() {
		return password;
	}

	public C getColor() {
		return color;
	}

	public void setNames(String[] names) {
		this.names = names;
	}

	public void setColor(C color) {
		this.color = color;
	}

}
